const { Client, LocalAuth, MessageMedia } = require("whatsapp-web.js");
const axios = require('axios');
const dayjs = require('dayjs');
const path = require('path');
const fs = require("fs");
const qrcode = require('qrcode-terminal');

// ================= CONFIGURAÇÕES E URLS (MANTIDAS) =================
const MEU_NUMERO_SIMULACAO = "5563984425197"; 
const GATILHO_TESTE = "JOHNTESTE"; 

const URL_WEBHOOK_HISTORICO = "https://script.google.com/macros/s/AKfycbyG30F-V52xN773jLyqDYdER0HzBoiYrvSjPc4lijgl2bOD-LnCR0xGtJi6JzhTFRi/exec";
const URL_PLANILHA_CSV = "https://docs.google.com/spreadsheets/d/e/2PACX-1vTXqSf5qMNA7Zd7jolV5IeplWYz-beU5-ypZHgmvUSlPxGIisq51hGbhHtlpnMf96OgG-TE4WIrLvKp/pub?gid=0&single=true&output=csv";

// ================= INICIALIZAÇÃO (MODO BACKGROUND) =================
const client = new Client({
    authStrategy: new LocalAuth({ clientId: "bot-hgp-v6", dataPath: path.join(__dirname, 'wwebjs_sessions') }),
    webVersionCache: { type: 'remote', remotePath: 'https://raw.githubusercontent.com/wppconnect-team/wa-version/main/html/2.2412.54.html' },
    puppeteer: {
        headless: true, // Para rodar com Windows bloqueado
        executablePath: 'C:\\Program Files\\Google\\Chrome\\Application\\chrome.exe',
        args: ["--no-sandbox", "--disable-setuid-sandbox", "--disable-dev-shm-usage", "--disable-gpu"]
    },
});

const estados = new Map();
const bloqueados = new Map();
const categoriasMap = { "1": "Soul MV", "2": "Impressora", "3": "Suporte Técnico", "4": "Telefonia / VOIP", "5": "Outras Solicitações" };
const delay = ms => new Promise(res => setTimeout(res, ms));

// ================= DISPARO DE MENSAGENS =================
async function enviarMensagemDireta(numeroBruto, mensagem) {
    try {
        let num = numeroBruto.replace(/\D/g, '');
        if (!num.startsWith('55')) num = '55' + num;
        let chatId = `${num}@c.us`;
        try {
            await client.sendMessage(chatId, mensagem);
            return true;
        } catch (e) {
            if (num.length === 13) {
                let semNono = num.slice(0, 4) + num.slice(5);
                await client.sendMessage(`${semNono}@c.us`, mensagem);
                return true;
            }
            throw e;
        }
    } catch (err) { return false; }
}

// ================= BLOQUEIO DE LIGAÇÕES (REATIVADO) =================
client.on('call', async (call) => {
    try {
        await call.reject();
        await client.sendMessage(call.from, "⚠️ *AVISO:* Este número é apenas para atendimento via chat (mensagens). Não recebemos chamadas de voz ou vídeo. Por favor, envie sua solicitação por texto.");
    } catch (e) { console.error("Erro ao rejeitar chamada."); }
});

// ================= BUSCA TÉCNICO ESCALA =================
async function buscarTecnicoEscala() {
    try {
        const res = await axios.get(URL_PLANILHA_CSV, { timeout: 10000 });
        const linhas = res.data.split(/\r?\n/).slice(1);
        const hojeISO = dayjs().format('YYYY-MM-DD');
        const hojeBR = dayjs().format('DD/MM/YYYY');
        for (let linha of linhas) {
            const col = linha.split(/,(?=(?:(?:[^"]*"){2})*[^"]*$)/).map(v => v.replace(/"/g,'').trim());
            if (col[0] === hojeISO || col[0] === hojeBR) return { nome: col[1], telefone: col[2] };
        }
        return null;
    } catch (e) { return null; }
}

// ================= FLUXO PRINCIPAL =================
client.on("message", async (msg) => {
    try {
        const chatId = msg.from;
        if (msg.fromMe || chatId.endsWith("@g.us")) return;
        const texto = msg.body.trim().toUpperCase();

        if (texto === "CANCELAR") { 
            estados.delete(chatId); 
            return await client.sendMessage(chatId, "❌ Atendimento encerrado."); 
        }

        let est = estados.get(chatId);

        if (texto === GATILHO_TESTE || !est) {
            const isTeste = (texto === GATILHO_TESTE);
            if (!isTeste && bloqueados.has(chatId) && Date.now() < bloqueados.get(chatId)) return;

            const contato = await msg.getContact();
            const saudacao = isTeste ? "🧪 *MODO SIMULAÇÃO*" : `*🛠️ TI - HGP*\nOlá, *${contato.pushname || 'Prezado'}*.`;
            
            await client.sendMessage(chatId, saudacao);
            await delay(500);
            await client.sendMessage(chatId, `1️⃣ Soul MV\n2️⃣ Impressora\n3️⃣ Suporte Técnico\n4️⃣ Telefonia / VOIP\n5️⃣ Outras\n6️⃣ Ramais\n\n_Ou envie CANCELAR._`);
            return estados.set(chatId, { step: 0.5, nomeWhats: contato.pushname || 'Prezado', isTeste: isTeste });
        }

        if (est.step === 0.5) {
            if (texto === "6") {
                const pdf = path.join(__dirname, 'RAMAIS TELEFÔNICOS - HGP.pdf');
                if (fs.existsSync(pdf)) await client.sendMessage(chatId, MessageMedia.fromFilePath(pdf));
                return estados.delete(chatId);
            }
            if (!categoriasMap[texto]) return;
            est.opcao = texto; est.step = 1;
            return await client.sendMessage(chatId, "👤 Seu *Nome Completo*:");
        }

        if (est.step === 1) { est.nome = texto; est.step = 2; return await client.sendMessage(chatId, "🏢 Seu *Setor e Ala*:"); }
        if (est.step === 2) { est.setor = texto; est.step = 3; return await client.sendMessage(chatId, "💻 *IP da Máquina*:"); }
        if (est.step === 3) { est.ip = texto; est.step = 4; return await client.sendMessage(chatId, "📱 *Telefone* de contato:"); }
        if (est.step === 4) { est.tel = texto; est.step = 5; return await client.sendMessage(chatId, "📝 Descreva o *Problema*:"); }

        if (est.step === 5) {
            est.desc = texto;
            const protocolo = `HGP-${dayjs().format('DDMM')}-${Math.random().toString(36).substring(2,5).toUpperCase()}`;
            const relatorio = `🛠️ *CHAMADO TI HGP*\n📌 *Protocolo:* ${protocolo}\n📂 *Categoria:* ${categoriasMap[est.opcao]}\n👤 *Solicitante:* ${est.nome}\n🏢 *Setor:* ${est.setor}\n💻 *IP:* ${est.ip}\n📱 *Contato:* ${est.tel}\n📝 *Problema:* ${est.desc}`;

            await client.sendMessage(chatId, relatorio);
            await client.sendMessage(chatId, "✅ Registrado e enviado ao técnico.");

            if (est.isTeste) {
                await enviarMensagemDireta(MEU_NUMERO_SIMULACAO, `🧪 *TESTE*:\n${relatorio}`);
            } else {
                const tecnico = await buscarTecnicoEscala();
                if (tecnico && tecnico.telefone) await enviarMensagemDireta(tecnico.telefone, `🚨 *NOVO CHAMADO REAL*\n\n${relatorio}`);
            }

            axios.post(URL_WEBHOOK_HISTORICO, { ...est, protocolo, categoria: categoriasMap[est.opcao] }).catch(() => {});
            if (!est.isTeste) bloqueados.set(chatId, Date.now() + (15 * 60 * 1000));
            estados.delete(chatId);
        }
    } catch (e) { console.error("Erro no fluxo."); }
});

client.on('qr', (qr) => qrcode.generate(qr, { small: true }));
client.on('ready', () => console.log('🚀 BOT HGP ATIVO - BACKGROUND E BLOQUEIO DE LIGAÇÕES ATIVOS'));
client.initialize();
